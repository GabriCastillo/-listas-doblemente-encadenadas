Class: DequeNodeTest
    DequeNode<Integer> firstNode, middleNode, lastNode

    getItem(middleNode) -> 2
    getItem(lastNode)   -> 3

    getNext(lastNode)   -> null

    getPrevious(lastNode)   -> middleNode

    isFirstNode(lastNode)   -> false

    isFirstNode(middleNode) -> false
    isFirstNode(lastNode)   -> true
